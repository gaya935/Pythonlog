import random
dice_cnt=[0,0,0,0,0,0]     # 리스트 초기화

for i in range(10000):
    number=random.randint(1,6)
    if number==1:            # 횟수 카운트
        dice_cnt[0]+=1       # elif는 중첩반복
    elif number==2:
        dice_cnt[1]+=1
    elif number==3:
        dice_cnt[2]+=1
    elif number==4:
        dice_cnt[3]+=1
    elif number==5:
        dice_cnt[4]+=1
    elif number==6:
        dice_cnt[5]+=1

print(dice_cnt)
